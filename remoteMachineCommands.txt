cd /home/pi/Images
echo "Load Image" >> BuildLogFile.txt
sudo docker load -i ./flowmodoro.tar >> BuildLogFile.txt

echo "remove container" >> BuildLogFile.txt
sudo docker stop Flowmodoro >> BuildLogFile.txt
sudo docker rm Flowmodoro >> BuildLogFile.txt

echo "start container" >> BuildLogFile.txt
sudo docker run -d -p 50000:80 -p 50001:443 --name Flowmodoro flowmodoro:latest >> BuildLogFile.txt

echo "DONE!" >> BuildLogFile.txt
truncate -s 10KB BuildLogFile.txt
